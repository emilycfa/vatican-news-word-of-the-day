  <script>

    const dateInput = document.getElementById("datePicker");

    const frame = document.getElementById("wordFrame");

    const urlDisplay = document.getElementById("urlDisplay");

    const goBtn = document.getElementById("goBtn");

    const todayBtn = document.getElementById("todayBtn");


    const TODAY_URL = "https://www.vaticannews.va/en/word-of-the-day.html";


    function pad2(num) {

      return num.toString().padStart(2, "0");

    }


    function setUrl(url) {

      frame.src = url;

      urlDisplay.textContent = url;

    }


    function setToToday() {

      const now = new Date(); // uses your local time

      const yyyy = now.getFullYear();

      const mm = pad2(now.getMonth() + 1);

      const dd = pad2(now.getDate());

      // Fill the date picker so you can adjust from "today"

      dateInput.value = `${yyyy}-${mm}-${dd}`;

      setUrl(TODAY_URL);

    }


    function loadSelectedDate() {

      const value = dateInput.value;

      if (!value) {

        // no date selected: just show today's main page

        setUrl(TODAY_URL);

        return;

      }


      const [year, month, day] = value.split("-");

      // Vatican News uses /YYYY/MM/DD.html for specific days

      const url = `https://www.vaticannews.va/en/word-of-the-day/${year}/${month}/${day}.html`;

      setUrl(url);

    }


    goBtn.addEventListener("click", loadSelectedDate);

    todayBtn.addEventListener("click", setToToday);


    // Initialize on first load

    setToToday();
