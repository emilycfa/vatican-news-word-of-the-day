    <script>

        const VATICAN_URL = 'https://www.vaticannews.va/en.html';

        const contentDiv = document.getElementById('content');


        async function fetchWordOfTheDay() {

            try {

                // NOTE: This fetch request is highly likely to be blocked by CORS (Cross-Origin Resource Sharing).

                // If it fails, you need to use a server-side proxy or a CORS-disabling browser extension.

                const response = await fetch(VATICAN_URL);

                if (!response.ok) {

                    throw new Error(`HTTP error! status: ${response.status}`);

                }

                

                // Since the content is not a simple JSON, we fetch the HTML text

                const htmlText = await response.text();

                

                // Create a temporary element to parse the HTML string

                const parser = new DOMParser();

                const doc = parser.parseFromString(htmlText, 'text/html');


                // --- CONTENT EXTRACTION LOGIC ---

                // The exact selector depends on the current structure of the Vatican News page.

                // This is a common pattern for "Daily Readings" on news sites.

                // You may need to inspect the Vatican News site and update the selector below.

                const wordElement = doc.querySelector('.article-box--reading-of-the-day, .article-box--gospel, [data-analytics-module="daily_reading"]');

                

                if (wordElement) {

                    // Display the entire relevant section found

                    contentDiv.innerHTML = wordElement.outerHTML;

                    // Remove any scripting or unwanted elements for cleanliness

                    contentDiv.querySelectorAll('script, style').forEach(el => el.remove());

                } else {

                    // Fallback: Display a message and a snippet of the page body

                    contentDiv.innerHTML = `<p class="error">Could not find the specific "Word of the Day" content element.</p>

                                            <p>The code may need updating to match the current Vatican News website structure.</p>

                                            <p>As a placeholder, here is the first part of the page text:</p>

                                            <hr>

                                            <p>${doc.body.textContent.substring(0, 500)}...</p>`;

                }


            } catch (error) {

                contentDiv.innerHTML = `<p class="error">Failed to load content.</p>

                                        <p>This is almost certainly due to **CORS (Cross-Origin Resource Sharing)** restrictions, which prevent a web page from fetching content from another domain.</p>

                                        <p>Error details: ${error.message}</p>

                                        <p>Please use a server-side proxy or local development environment to resolve this.</p>`;

                console.error('Fetch error:', error);

            }

        }


        fetchWordOfTheDay();

    </script>
